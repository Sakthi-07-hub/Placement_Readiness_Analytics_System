CREATE DATABASE IF NOT EXISTS placement_readiness;
USE placement_readiness;
-- ------------------------------------------------------------
-- TABLE 1 : students
-- Core student profile — one row per student
-- ------------------------------------------------------------
CREATE TABLE students (
    student_id      INT AUTO_INCREMENT PRIMARY KEY,
    full_name       VARCHAR(100)   NOT NULL,
    email           VARCHAR(100)   NOT NULL UNIQUE,
    phone           VARCHAR(15),
    branch          ENUM('CSE','IT','ECE','EEE','MECH','CIVIL') NOT NULL,
    year_of_study   TINYINT        NOT NULL CHECK (year_of_study BETWEEN 1 AND 4),
    cgpa            DECIMAL(4,2)   NOT NULL CHECK (cgpa BETWEEN 0.00 AND 10.00),
    active_backlogs TINYINT        DEFAULT 0,
    gender          ENUM('Male','Female','Other'),
    city            VARCHAR(60),
    enrolled_on     DATE           DEFAULT (CURRENT_DATE),
    INDEX idx_branch (branch),
    INDEX idx_cgpa   (cgpa)
);

-- ------------------------------------------------------------
-- TABLE 2 : skills
-- Technical & soft skills with self-rated proficiency (1–5)
-- ------------------------------------------------------------
CREATE TABLE skills (
    skill_id        INT AUTO_INCREMENT PRIMARY KEY,
    student_id      INT            NOT NULL,
    skill_name      VARCHAR(80)    NOT NULL,
    skill_category  ENUM('Technical','Soft','Domain') NOT NULL,
    proficiency     TINYINT        NOT NULL CHECK (proficiency BETWEEN 1 AND 5),
    FOREIGN KEY (student_id) REFERENCES students(student_id) ON DELETE CASCADE,
    INDEX idx_student_skill (student_id)
);

-- ------------------------------------------------------------
-- TABLE 3 : certifications
-- Online / offline certifications completed
-- ------------------------------------------------------------
CREATE TABLE certifications (
    cert_id         INT AUTO_INCREMENT PRIMARY KEY,
    student_id      INT            NOT NULL,
    cert_name       VARCHAR(120)   NOT NULL,
    issuing_org     VARCHAR(80),
    completion_date DATE,
    score_percent   DECIMAL(5,2)   CHECK (score_percent BETWEEN 0 AND 100),
    FOREIGN KEY (student_id) REFERENCES students(student_id) ON DELETE CASCADE,
    INDEX idx_student_cert (student_id)
);

-- ------------------------------------------------------------
-- TABLE 4 : internships
-- Internship experience (duration in weeks)
-- ------------------------------------------------------------
CREATE TABLE internships (
    intern_id       INT AUTO_INCREMENT PRIMARY KEY,
    student_id      INT            NOT NULL,
    company_name    VARCHAR(100)   NOT NULL,
    role            VARCHAR(100),
    duration_weeks  TINYINT        NOT NULL CHECK (duration_weeks > 0),
    stipend         DECIMAL(8,2)   DEFAULT 0,
    is_paid         BOOLEAN        DEFAULT FALSE,
    start_date      DATE,
    end_date        DATE,
    FOREIGN KEY (student_id) REFERENCES students(student_id) ON DELETE CASCADE,
    INDEX idx_student_intern (student_id)
);

-- ------------------------------------------------------------
-- TABLE 5 : projects
-- Academic or personal projects with complexity level
-- ------------------------------------------------------------
CREATE TABLE projects (
    project_id      INT AUTO_INCREMENT PRIMARY KEY,
    student_id      INT            NOT NULL,
    project_title   VARCHAR(150)   NOT NULL,
    tech_stack      VARCHAR(200),
    complexity      ENUM('Basic','Intermediate','Advanced') NOT NULL,
    has_github      BOOLEAN        DEFAULT FALSE,
    is_deployed     BOOLEAN        DEFAULT FALSE,
    completion_date DATE,
    FOREIGN KEY (student_id) REFERENCES students(student_id) ON DELETE CASCADE,
    INDEX idx_student_project (student_id)
);

-- ------------------------------------------------------------
-- TABLE 6 : mock_tests
-- Aptitude, Coding, and HR mock interview scores (0–100)
-- ------------------------------------------------------------
CREATE TABLE mock_tests (
    test_id         INT AUTO_INCREMENT PRIMARY KEY,
    student_id      INT            NOT NULL,
    test_type       ENUM('Aptitude','Coding','Communication','HR') NOT NULL,
    score           DECIMAL(5,2)   NOT NULL CHECK (score BETWEEN 0 AND 100),
    max_score       DECIMAL(5,2)   DEFAULT 100,
    test_date       DATE,
    attempt_number  TINYINT        DEFAULT 1,
    FOREIGN KEY (student_id) REFERENCES students(student_id) ON DELETE CASCADE,
    INDEX idx_student_test (student_id),
    INDEX idx_test_type    (test_type)
);

-- ------------------------------------------------------------
-- TABLE 7 : placement_score  (computed — populated by VIEW / SP)
-- Stores the final weighted Placement Readiness Score per student
-- ------------------------------------------------------------
CREATE TABLE placement_score (
    score_id              INT AUTO_INCREMENT PRIMARY KEY,
    student_id            INT            NOT NULL UNIQUE,
    academic_score        DECIMAL(5,2)   DEFAULT 0  COMMENT 'Max 25 pts',
    skills_score          DECIMAL(5,2)   DEFAULT 0  COMMENT 'Max 20 pts',
    certifications_score  DECIMAL(5,2)   DEFAULT 0  COMMENT 'Max 10 pts',
    internship_score      DECIMAL(5,2)   DEFAULT 0  COMMENT 'Max 20 pts',
    projects_score        DECIMAL(5,2)   DEFAULT 0  COMMENT 'Max 15 pts',
    mock_test_score       DECIMAL(5,2)   DEFAULT 0  COMMENT 'Max 10 pts',
    total_score           DECIMAL(5,2)   DEFAULT 0  COMMENT 'Max 100 pts',
    readiness_band        ENUM('High','Medium','Low') DEFAULT 'Low',
    last_updated          TIMESTAMP      DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    FOREIGN KEY (student_id) REFERENCES students(student_id) ON DELETE CASCADE
);

-- ============================================================
--  PLACEMENT READINESS ANALYTICS SYSTEM
--  Phase 3 : Placement Readiness Score Engine (MySQL)
--  Weighted Scoring via Views + Stored Procedure
-- ============================================================
 
USE placement_readiness;
 
-- ============================================================
-- SCORING FORMULA REFERENCE
-- ============================================================
--  Component         Weight   Max Pts   Logic
--  ──────────────    ──────   ───────   ─────────────────────────────────
--  Academic (CGPA)    25%      25        (CGPA / 10) * 25
--  Skills             20%      20        MIN(tech_skill_count * avg_prof / 25, 1) * 20
--  Certifications     10%      10        MIN(cert_count, 5) / 5 * 10
--  Internships        20%      20        MIN(total_weeks / 24, 1) * 15 + paid_bonus (5)
--  Projects           15%      15        complexity points + github/deploy bonus
--  Mock Tests         10%      10        (avg_mock_score / 100) * 10
--  ──────────────    ──────   ───────
--  TOTAL             100%     100
--
--  Readiness Band:
--    High   : total_score >= 70
--    Medium : total_score >= 40 AND < 70
--    Low    : total_score < 40
-- ============================================================

-- STEP 1 : VIEW — v_academic_score
-- Calculates academic component out of 25 pts
-- Penalises active backlogs
-- ============================================================

CREATE OR REPLACE VIEW v_academic_score AS 
SELECT
	student_id,
    cgpa,
    active_backlogs,
    ROUND(
		GREATEST(
			((cgpa / 10.0 )* 25) - (active_backlogs * 2),
            0
		), 2
	) AS academic_score
FROM students;

SELECT * FROM v_academic_score;

-- ============================================================
-- STEP 2 : VIEW — v_skills_score
-- Calculates skills component out of 20 pts
-- Based on number of Technical skills + average proficiency
-- ============================================================

CREATE OR REPLACE VIEW v_skills_score AS 
SELECT
	s.student_id,
	COUNT(CASE WHEN sk.skill_category = "Technical" THEN 1 END) AS tech_skill_count,
    ROUND(AVG(sk.proficiency) , 2) AS avg_proficiency,
    ROUND(
		LEAST(
			(COUNT(CASE WHEN sk.skill_category = "Technical" THEN 1 END) *
            COALESCE(AVG(sk.proficiency) , 0)) / 25.0 * 20,
			20
		) , 2
	) AS skills_score
FROM students s
LEFT JOIN skills sk 
ON s.student_id = sk.student_id
GROUP BY s.student_id;

SELECT * FROM v_skills_score;

-- ============================================================
-- STEP 3 : VIEW — v_certifications_score
-- Calculates certifications component out of 10 pts
-- Max 5 certs counted; each worth 2 pts
-- ============================================================

CREATE OR REPLACE VIEW v_certifications_score AS
SELECT 
	s.student_id ,
    COUNT(c.cert_id) AS cert_count,
    ROUND(
		LEAST(COUNT(c.cert_id) , 5 ) / 5.0 * 10,
        2
	) AS certifications_score
FROM students s
LEFT JOIN certifications c 
ON s.student_id = c.student_id
GROUP BY student_id;

SELECT * FROM v_certifications_score;

-- ============================================================
-- STEP 4 : VIEW — v_internship_score
-- Calculates internship component out of 20 pts
-- Duration worth 15 pts (capped at 24 weeks)
-- Paid internship bonus: 5 pts
-- ============================================================

CREATE OR REPLACE VIEW v_internship_score AS
SELECT 
	s.student_id,
    COALESCE(SUM(i.duration_weeks), 0) AS total_weeks,
    MAX(
		CASE
			WHEN i.is_paid = TRUE THEN 1
			ELSE 0
	    END
	)                     AS has_paid_internship,
    
    ROUND(
		LEAST(
			COALESCE(SUM(i.duration_weeks) , 0) / 24.0 , 1) * 15
            + (MAX(CASE WHEN i.is_paid = TRUE THEN 1 ELSE 0 END) * 5),
            2
		) AS internship_score
FROM students s
LEFT JOIN internships i 
ON s.student_id = i.student_id
GROUP BY student_id;

SELECT * FROM v_internship_score;

-- ============================================================
-- STEP 5 : VIEW — v_projects_score
-- Calculates projects component out of 15 pts
-- Basic=1pt, Intermediate=2pt, Advanced=3pt per project
-- GitHub bonus: 0.5pt per project; Deploy bonus: 0.5pt per project
-- Capped at 15 pts
-- ============================================================

CREATE OR REPLACE VIEW v_projects_score AS 
SELECT
	s.student_id,
    COUNT(p.project_id) AS project_count,
    ROUND(	
		LEAST(
			COALESCE(
				SUM(
					CASE p.complexity
						WHEN "Basic" THEN 1.0
						WHEN "Intermediate" THEN 1.0
						WHEN "Advanced" THEN 1.0
						ELSE 0
                    END
                    + (CASE WHEN p.has_github = TRUE THEN 0.5 ELSE 0 END) 
                    + (CASE WHEN p.is_deployed = TRUE THEN 0.5 ELSE 0 END)
				),0
			), 15
		),2
	) AS projects_score
FROM students s
LEFT JOIN projects p 
ON s.student_id = p.student_id
GROUP BY s.student_id;
                    
SELECT * FROM projects_score;

-- ============================================================
-- STEP 6 : VIEW — v_mock_test_score
-- Calculates mock test component out of 10 pts
-- Average of all test scores normalised to 10
-- ============================================================

CREATE OR REPLACE VIEW v_mock_test_score AS 
SELECT 
	s.student_id,
    ROUND(AVG(mt.score),2) AS avg_mock_score,
    ROUND(COALESCE(AVG(mt.score),0) / 100 * 10 , 2) AS mock_test_score
FROM students s
LEFT JOIN mock_tests mt 
ON s.student_id = mt.student_id
GROUP BY s.student_id;

-- ============================================================
-- STEP 7 : MASTER VIEW — v_placement_readiness
-- Joins all 6 component views into one master readiness view
-- This is the view you will connect to Power BI
-- ============================================================

CREATE OR REPLACE VIEW v_placement_readiness AS
SELECT
    st.student_id,
    st.full_name,
    st.branch,
    st.year_of_study,
    st.cgpa,
    st.active_backlogs,
    st.gender,
    st.city,
    
     -- Component scores
    ac.academic_score,
    sk.skills_score,
    ce.certifications_score,
    iv.internship_score,
    pj.projects_score,
    mt.mock_test_score,
    
    -- Total score

	ROUND(
		 ac.academic_score+
		sk.skills_score+
		ce.certifications_score+
		iv.internship_score+
		pj.projects_score+
		mt.mock_test_score,
        2
    ) AS Total_score,
		
        -- Readiness band

CASE 
	WHEN 
		ROUND(
			 ac.academic_score + sk.skills_score + ce.certifications_score +
             iv.internship_score + pj.projects_score + mt.mock_test_score, 2
	) >= 70 THEN 'High'
    
    WHEN 
		ROUND(
			 ac.academic_score + sk.skills_score + ce.certifications_score +
             iv.internship_score + pj.projects_score + mt.mock_test_score, 2
	) >= 40 THEN 'Medium'
    ELSE 'Low'
END AS Readiness_band,

    -- Supporting counts for dashboards
    
    sk.tech_skill_count,
    sk.avg_proficiency,
    ce.cert_count,
    iv.total_weeks        AS internship_weeks,
    iv.has_paid_internship,
    pj.project_count,
    mt.avg_mock_score
    
FROM students st
JOIN v_academic_score      ac ON st.student_id = ac.student_id
JOIN v_skills_score        sk ON st.student_id = sk.student_id
JOIN v_certifications_score ce ON st.student_id = ce.student_id
JOIN v_internship_score    iv ON st.student_id = iv.student_id
JOIN v_projects_score      pj ON st.student_id = pj.student_id
JOIN v_mock_test_score     mt ON st.student_id = mt.student_id;

SELECT * FROM v_placement_readiness;

-- ============================================================
-- STEP 8 : STORED PROCEDURE — sp_calculate_placement_scores
-- Reads from v_placement_readiness and populates placement_score table
-- Run this whenever student data changes
-- ============================================================

DROP PROCEDURE IF EXISTS sp_calculate_placement_scores;
 
DELIMITER $$
 
CREATE PROCEDURE sp_calculate_placement_scores()
BEGIN
    -- Clear existing scores
    TRUNCATE TABLE placement_score;
 
    -- Insert fresh computed scores for all students
    INSERT INTO placement_score (
        student_id,
        academic_score,
        skills_score,
        certifications_score,
        internship_score,
        projects_score,
        mock_test_score,
        total_score,
        readiness_band
    )
    SELECT
        student_id,
        academic_score,
        skills_score,
        certifications_score,
        internship_score,
        projects_score,
        mock_test_score,
        total_score,
        readiness_band
    FROM v_placement_readiness;
 
    -- Confirmation message
    SELECT
        COUNT(*)                                      AS total_students_scored,
        SUM(CASE WHEN readiness_band = 'High'   THEN 1 ELSE 0 END) AS high_band,
        SUM(CASE WHEN readiness_band = 'Medium' THEN 1 ELSE 0 END) AS medium_band,
        SUM(CASE WHEN readiness_band = 'Low'    THEN 1 ELSE 0 END) AS low_band,
        ROUND(AVG(total_score), 2)                    AS avg_score,
        ROUND(MAX(total_score), 2)                    AS top_score,
        ROUND(MIN(total_score), 2)                    AS lowest_score
    FROM placement_score;
END$$
 
DELIMITER ;

-- ============================================================
-- STEP 9 : EXECUTE — Run the stored procedure
USE placement_readiness;

CALL sp_calculate_placement_scores();

-- ============================================================
-- QUERY 02 : Readiness Band Distribution
-- Purpose  : How many students fall in High / Medium / Low
-- Use in   : Donut / Pie chart in Power BI
-- ============================================================

SELECT 
	readiness_band,
    COUNT(*) AS student_count,
    ROUND(COUNT(*) * 100.0 / SUM(COUNT(*)) OVER() , 1) AS percentage
FROM v_placement_readiness
GROUP BY readiness_band
ORDER BY FIELD(readiness_band , 'High' , 'Medium' ,'Low' );

-- ============================================================
-- QUERY 03 : Branch-wise Average Score Comparison
-- Purpose  : Which branch performs best overall?
-- Use in   : Bar chart in Power BI
-- ============================================================

SELECT
    branch,
    COUNT(*)                          AS total_students,
    ROUND(AVG(cgpa), 2)               AS avg_cgpa,
    ROUND(AVG(academic_score), 2)     AS avg_academic,
    ROUND(AVG(skills_score), 2)       AS avg_skills,
    ROUND(AVG(internship_score), 2)   AS avg_internship,
    ROUND(AVG(projects_score), 2)     AS avg_projects,
    ROUND(AVG(mock_test_score), 2)    AS avg_mock,
    ROUND(AVG(total_score), 2)        AS avg_total_score
FROM v_placement_readiness
GROUP BY branch
ORDER BY avg_total_score DESC;

    cgpa,
    total_score,
    readiness_band,

    internship_weeks,

LIMIT 10; 

-- ============================================================
-- QUERY 05 : Students Needing Immediate Attention (Low Band)
-- Purpose  : Identify at-risk students for intervention
-- Use in   : Alert table / conditional formatting in Power BI
-- ============================================================

SELECT
    student_id,
    full_name,
    branch,
    year_of_study,
    cgpa,
    active_backlogs,
    total_score,
    ROUND(100 - total_score, 2)            AS score_gap_to_max,
    CASE 
		WHEN academic_score < 10 THEN 'Improve CGPA'
        WHEN skills_score < 8 THEN 'Learn More Skills'
        WHEN internship_score = 0 THEN 'Get an Internship'
        WHEN cert_count = 0 THEN 'Earn Certifications'
        ELSE 'General Improvement'
	END AS 'Top Recommendations'
FROM v_placement_readiness
WHERE readiness_band = 'Low'
ORDER BY total_score ASC;

-- ============================================================
-- QUERY 06 : Component-wise Score Gap Analysis
-- Purpose  : Find the weakest scoring area across all students
-- Use in   : Radar / Spider chart in Power BI
-- ============================================================

SELECT
    'Academic'       AS component, ROUND(AVG(academic_score), 2)      AS avg_score, 25  AS max_score, ROUND(AVG(academic_score)/25*100, 1)      AS pct_achieved FROM v_placement_readiness
UNION ALL
SELECT 'Skills',                   ROUND(AVG(skills_score), 2),        20,           ROUND(AVG(skills_score)/20*100, 1)        FROM v_placement_readiness
UNION ALL
SELECT 'Certifications',           ROUND(AVG(certifications_score), 2),10,           ROUND(AVG(certifications_score)/10*100, 1)FROM v_placement_readiness
UNION ALL
SELECT 'Internships',              ROUND(AVG(internship_score), 2),    20,           ROUND(AVG(internship_score)/20*100, 1)    FROM v_placement_readiness
UNION ALL
SELECT 'Projects',                 ROUND(AVG(projects_score), 2),      15,           ROUND(AVG(projects_score)/15*100, 1)      FROM v_placement_readiness
UNION ALL
SELECT 'Mock Tests',               ROUND(AVG(mock_test_score), 2),     10,           ROUND(AVG(mock_test_score)/10*100, 1)     FROM v_placement_readiness
ORDER BY pct_achieved ASC;
    
-- ============================================================
-- QUERY 07 : Gender-wise Readiness Comparison
-- Purpose  : Diversity and inclusion insight
-- Use in   : Grouped bar chart in Power BI
-- ============================================================

SELECT
    gender,
    COUNT(*)                           AS total_students,
    ROUND(AVG(total_score), 2)         AS avg_score,
    SUM(CASE WHEN readiness_band = 'High'   THEN 1 ELSE 0 END) AS high_count,
    SUM(CASE WHEN readiness_band = 'Medium' THEN 1 ELSE 0 END) AS medium_count,
    SUM(CASE WHEN readiness_band = 'Low'    THEN 1 ELSE 0 END) AS low_count
FROM v_placement_readiness
GROUP BY gender;    

-- ============================================================
-- QUERY 08 : Year-of-Study vs Readiness Progression
-- Purpose  : Do senior students score higher? (Trend analysis)
-- Use in   : Line chart in Power BI
-- ============================================================

SELECT
    year_of_study,
    COUNT(*)                           AS student_count,
    ROUND(AVG(total_score), 2)         AS avg_total_score,
    ROUND(AVG(cgpa), 2)                AS avg_cgpa,
    ROUND(AVG(tech_skill_count), 1)    AS avg_skills,
    ROUND(AVG(cert_count), 1)          AS avg_certs,
    ROUND(AVG(internship_weeks), 1)    AS avg_intern_weeks,
    ROUND(AVG(project_count), 1)       AS avg_projects
FROM v_placement_readiness
GROUP BY year_of_study
ORDER BY year_of_study;

-- ============================================================
-- QUERY 09 : Internship Impact on Placement Score
-- Purpose  : Do students with paid internships score higher?
-- Use in   : Comparison card / bar chart in Power BI
-- ============================================================

SELECT
    CASE
        WHEN has_paid_internship = 1 AND internship_weeks >= 12 THEN 'Paid + Long (12w+)'
        WHEN has_paid_internship = 1 AND internship_weeks  < 12 THEN 'Paid + Short (<12w)'
        WHEN has_paid_internship = 0 AND internship_weeks  > 0  THEN 'Unpaid Internship'
        ELSE 'No Internship'
    END AS internship_category,
    COUNT(*)                           AS student_count,
    ROUND(AVG(total_score), 2)         AS avg_total_score,
    ROUND(AVG(internship_score), 2)    AS avg_intern_score
FROM v_placement_readiness
GROUP BY internship_category
ORDER BY avg_total_score DESC;

-- ============================================================
-- QUERY 10 : Top Skills Among High-Band Students
-- Purpose  : What skills do placement-ready students have?
-- Use in   : Word cloud / horizontal bar in Power BI
-- ============================================================

SELECT
    sk.skill_name,
    sk.skill_category,
    COUNT(*)                           AS student_count,
    ROUND(AVG(sk.proficiency), 2)      AS avg_proficiency
FROM skills sk
JOIN v_placement_readiness vpr ON sk.student_id = vpr.student_id
WHERE vpr.readiness_band = 'High'
  AND sk.skill_category  = 'Technical'
GROUP BY sk.skill_name, sk.skill_category
ORDER BY student_count DESC, avg_proficiency DESC
LIMIT 15;
